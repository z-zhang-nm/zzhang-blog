归并排序

四月 11, 2021 发布在 算法, 排序算法

　　分治的方法，“分”即将问题分为一些小的问题递归求解，“治”即将各个小问题合在一起

算法思想

递归将数组分为子序列，递归深度为$log_2(N)$，对两个子数组进行归并排序后合并两个数组，合并时用双指针的方法。

算法实现

class Sort {
 public:
  explicit Sort(vector<int> &arr) : nums(arr) {}
  void mergeSort(size_t left, size_t right);
  void printNums();

 private:
  vector<int> nums;
  void swap(size_t i, size_t j);
};

void Sort::swap(size_t i, size_t j) {
  int temp = nums[i];
  nums[i] = nums[j];
  nums[j] = temp;
}

void Sort::printNums() {
  for (auto &num : nums) {
    cout << num << " ";
  }
  cout << endl;
}

void Sort::mergeSort(size_t left, size_t right) {
  if (left == right) return;
  if (left + 1 == right) {
    if (nums[left] > nums[right]) {
      swap(left, right);
    }
  }
  vector<int> ret(right - left + 1, 0);
  size_t mid = (right - left) / 2 + left;
  mergeSort(left, mid);
  mergeSort(mid + 1, right);
  size_t i = left, j = mid + 1, k = 0;
  while (i <= mid && j <= right) {
    ret[k++] = (nums[i] <= nums[j]) ? nums[i++] : nums[j++];
  }
  while (i <= mid) {
    ret[k++] = nums[i++];
  }
  while (j <= right) {
    ret[k++] = nums[j++];
  }
  for (size_t i = left; i <= right; i++) {
    nums[i] = ret[i - left];
  }
}

int main() {
  vector<int> nums{3, 6, 1, 2, 8, 9, 5, 7, 4};
  Sort sort(nums);
  sort.mergeSort(0, nums.size() - 1);
  sort.printNums();
}

上面方法会在每次递归时都申请新的空间，效率较低，可以在一开始就申请好和数组一样大的空间，且left + 1 == right的判断也没有必要。

class Sort {
 public:
  explicit Sort(vector<int> &arr) : nums(arr) {}
  void mergeSort();
  void printNums();

 private:
  vector<int> nums;
  void swap(size_t i, size_t j);
  void merge(size_t left, size_t right, vector<int> &temp);
};

void Sort::swap(size_t i, size_t j) {
  int temp = nums[i];
  nums[i] = nums[j];
  nums[j] = temp;
}

void Sort::printNums() {
  for (auto &num : nums) {
    cout << num << " ";
  }
  cout << endl;
}

void Sort::mergeSort() {
  assert(nums.size());
  vector<int> temp(nums.size(), 0);
  merge(0, nums.size() - 1, temp);
  printNums();
}

void Sort::merge(size_t left, size_t right, vector<int> &temp) {
  if (left == right) return;
  size_t mid = (right - left) / 2 + left;
  merge(left, mid, temp);
  merge(mid + 1, right, temp);
  size_t i = left, j = mid + 1, k = 0;
  while (i <= mid && j <= right) {
    temp[k++] = (nums[i] <= nums[j]) ? nums[i++] : nums[j++];
  }
  while (i <= mid) {
    temp[k++] = nums[i++];
  }
  while (j <= right) {
    temp[k++] = nums[j++];
  }
  k = 0;
  for (size_t i = left; i <= right; i++, k++) {
    nums[i] = temp[k];
  }
}

int main() {
  vector<int> nums{3, 6, 1, 2, 8, 9, 5, 7, 4};
  Sort sort(nums);
  sort.mergeSort();
}

复杂度分析

由于递归调用层数为$log_2(N)$，每一层需要比较排序的个数都是N，因此整体时间复杂度为$N*log_2(N)$。 空间复杂度为O(N)。 稳定排序。
