# apt-get 安装 pip
- pythoh2: `sudo apt-get install python-pip`
- pythoh3: `sudo apt-get install python3-pip`

# pip 安装组件
- openpyxl: `pip install openpyxl==2.6.4`
- pandas,numpy: `pip install pandas==0.23.0 numpy==1.11.0`
